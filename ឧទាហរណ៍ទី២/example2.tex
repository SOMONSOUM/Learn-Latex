\documentclass[a4paper, 12pt]{article}
\usepackage{amsmath}

\usepackage{amsthm}
\theoremstyle{definition}
\newtheorem{theorem}{Theorem}
\begin{document}
	\begin{theorem}
		For real numbers $ a,b\in\mathbb{R} $ we have 
		$$ a^2+b^2 \ge 2ab $$
	\end{theorem}
	\begin{proof}
		For $ a,b\in\mathbb{R} $, we have $ (a-b)^2 \ge 0 $ and therefore
		$$ a^2+b^2 \ge 2ab $$
	\end{proof}
\end{document}
